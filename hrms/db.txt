create database emp_management_sys;

use emp_management_sys;

CREATE TABLE employee (
    employee_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name VARCHAR(255),
    category_of_employment VARCHAR(100),
    designation VARCHAR(100),
    department VARCHAR(100),
    reporting_to VARCHAR(255),
    date_of_joining DATE,
    month VARCHAR(50),
    years_of_completion FLOAT,
    approx_years_completion INT,
    probation_period INT,
    date_of_confirmation DATE,
    location VARCHAR(255),
    date_of_birth DATE,
    age_till_date INT,
    date_of_month INT,
    appointment_letter VARCHAR(255),
    nda VARCHAR(255),
    confirmation_or_probation VARCHAR(50),
    id_card VARCHAR(100),
    account VARCHAR(100),
    bank_account VARCHAR(100),
    ifsc_code VARCHAR(100),
    uan_number VARCHAR(100),
    pf_number VARCHAR(100),
    marital_status VARCHAR(50),
    gender VARCHAR(10),
    official_contact_no VARCHAR(20),
    personal_contact_no VARCHAR(20),
    alternate_contact_no VARCHAR(20),
    personal_email_id VARCHAR(255),
    official_email_id VARCHAR(255),
    blood_group VARCHAR(10),
    current_address TEXT,
    permanent_address TEXT,
    highest_qualification VARCHAR(255),
    total_experience INT,
    relevant_experience INT,
    previous_company VARCHAR(255),
    designation_in_previous_company VARCHAR(100),
    experience_with_ppt VARCHAR(50),
    date_of_resignation DATE,
    retirement_date DATE,
    exit_date DATE,
    education_certificate VARCHAR(255),
    ration_card VARCHAR(255),
    pan_card VARCHAR(255),
    aadhar_card VARCHAR(255),
    driving_license VARCHAR(255),
    passport VARCHAR(255),
    bank_name VARCHAR(255),
    name_as_per_bank VARCHAR(255),
    account_number VARCHAR(100),
    ifsc_code_bank VARCHAR(100)
);

CREATE TABLE addresses (
  address_id INT AUTO_INCREMENT NOT NULL,
  employee_id INT NOT NULL,
  address_line1 VARCHAR(100) NOT NULL,
  address_line2 VARCHAR(100) DEFAULT NULL,
  city VARCHAR(50) NOT NULL,
  state VARCHAR(50) NOT NULL,
  country VARCHAR(50) NOT NULL,
  pin_code VARCHAR(10) NOT NULL,
  created_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (address_id)
);

CREATE TABLE bank_details (
  bank_detail_id INT AUTO_INCREMENT NOT NULL,
  employee_id INT NOT NULL,
  bank_name VARCHAR(100) NOT NULL,
  bank_acc_no VARCHAR(50) NOT NULL,
  ifsc_no VARCHAR(20) NOT NULL,
  upi_id VARCHAR(50) NOT NULL,
  created_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (bank_detail_id)
);



CREATE TABLE roles (
  role_id INT AUTO_INCREMENT NOT NULL,
  role_name VARCHAR(50) NOT NULL,
  role_level INT NOT NULL,
  role_reporting_to INT DEFAULT NULL,
  created_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (role_id)
);
