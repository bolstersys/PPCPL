create database emp_management_sys;

use emp_management_sys;

CREATE TABLE employee (
    employee_id NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name VARCHAR2(255),
    category_of_employment VARCHAR2(100),
    designation VARCHAR2(100),
    department VARCHAR2(100),
    reporting_to VARCHAR2(255),
    date_of_joining DATE,
    month VARCHAR2(50),
    years_of_completion FLOAT,
    approx_years_completion NUMBER,
    probation_period NUMBER,
    date_of_confirmation DATE,
    location VARCHAR2(255),
    date_of_birth DATE,
    age_till_date NUMBER,
    date_of_month NUMBER,
    appointment_letter VARCHAR2(255),
    nda VARCHAR2(255),
    confirmation_or_probation VARCHAR2(50),
    id_card VARCHAR2(100),
    account VARCHAR2(100),
    bank_account VARCHAR2(100),
    ifsc_code VARCHAR2(100),
    uan_number VARCHAR2(100),
    pf_number VARCHAR2(100),
    marital_status VARCHAR2(50),
    gender VARCHAR2(10),
    official_contact_no VARCHAR2(20),
    personal_contact_no VARCHAR2(20),
    alternate_contact_no VARCHAR2(20),
    personal_email_id VARCHAR2(255),
    official_email_id VARCHAR2(255),
    blood_group VARCHAR2(10),
    current_address CLOB,
    permanent_address CLOB,
    highest_qualification VARCHAR2(255),
    total_experience NUMBER,
    relevant_experience NUMBER,
    previous_company VARCHAR2(255),
    designation_in_previous_company VARCHAR2(100),
    experience_with_ppt VARCHAR2(50),
    date_of_resignation DATE,
    retirement_date DATE,
    exit_date DATE,
    education_certificate VARCHAR2(255),
    ration_card VARCHAR2(255),
    pan_card VARCHAR2(255),
    aadhar_card VARCHAR2(255),
    driving_license VARCHAR2(255),
    passport VARCHAR2(255),
    bank_name VARCHAR2(255),
    name_as_per_bank VARCHAR2(255),
    account_number VARCHAR2(100),
    ifsc_code_bank VARCHAR2(100)
);


CREATE TABLE addresses (
  address_id INT AUTO_INCREMENT NOT NULL,
  employee_id INT NOT NULL,
  address_line1 VARCHAR(100) NOT NULL,
  address_line2 VARCHAR(100) DEFAULT NULL,
  city VARCHAR(50) NOT NULL,
  state VARCHAR(50) NOT NULL,
  country VARCHAR(50) NOT NULL,
  pin_code VARCHAR(10) NOT NULL,
  created_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (address_id)
);

CREATE TABLE bank_details (
  bank_detail_id INT AUTO_INCREMENT NOT NULL,
  employee_id INT NOT NULL,
  bank_name VARCHAR(100) NOT NULL,
  bank_acc_no VARCHAR(50) NOT NULL,
  ifsc_no VARCHAR(20) NOT NULL,
  upi_id VARCHAR(50) NOT NULL,
  created_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (bank_detail_id)
);



CREATE TABLE role (
  role_code NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  role_name VARCHAR(50) NOT NULL,
  role_level INT NOT NULL
);
